# panos_decryption_profile (Resource)

Manages a PAN-OS **decryption profile** on Panorama. Decryption profiles define settings that control how the firewall handles SSL/TLS traffic that has been decrypted, including certificate validation, protocol version restrictions, and cipher suite controls.

## Example Usage

### Panorama — Shared

```hcl
resource "panos_decryption_profile" "example" {
  location = {
    shared = {}
  }
  name = "corp-decrypt-profile"

  ssl_forward_proxy = {
    block_expired_certificate = true
    block_untrusted_issuer    = true
    block_unknown_cert        = true
    block_unsupported_version = true
    block_unsupported_cipher  = true
  }

  ssl_protocol_settings = {
    min_version = "tls1-2"
    max_version = "tls1-3"

    enc_algo_aes_128_gcm = true
    enc_algo_aes_256_gcm = true
    enc_algo_aes_128_cbc = true
    enc_algo_aes_256_cbc = true

    keyxchg_algo_ecdhe = true
    keyxchg_algo_dhe   = true

    auth_algo_sha256 = true
    auth_algo_sha384 = true
  }
}
```

### Panorama — Device Group

```hcl
resource "panos_decryption_profile" "dg_example" {
  location = {
    device_group = {
      name            = "my-device-group"
      panorama_device = "localhost.localdomain"
    }
  }
  name = "inbound-inspect-profile"

  ssl_inbound_inspection = {
    block_unsupported_version         = true
    block_unsupported_cipher          = true
    block_if_no_resource              = true
    block_if_hsm_unavailable          = true
    block_tls13_downgrade_no_resource = true
  }

  ssl_protocol_settings = {
    min_version = "tls1-2"
    max_version = "tls1-3"
  }
}
```
